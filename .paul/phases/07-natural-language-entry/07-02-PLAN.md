    **2. Create `LoomCal/Services/NLParseService.swift`:**

    An `@MainActor` class (NOT ObservableObject — used as a utility, not a ViewModel):

    ```swift
    @MainActor
    final class NLParseService {
        static let shared = NLParseService()

        /// Submit an NL parse request and wait for the result.
        /// Returns the parsed result, or nil on error/timeout.
        func parse(text: String, type: String) async -> ParsedNLResult? {
            let requestId = UUID().uuidString

            // 1. Create parse request in Convex
            let args: [String: ConvexEncodable?] = [
                "requestId": requestId,
                "text": text,
                "type": type,
            ]
            try? await convex.mutation("nlParse:createParseRequest", with: args)

            // 2. Subscribe to result and wait for completion or timeout
            // Use a Task with timeout
            return await withTimeout(seconds: 10) {
                for await results: [ParsedNLResult] in convex
                    .subscribe(to: "nlParse:getResult", with: ["requestId": requestId])
                    .replaceError(with: [])
                    .values {
                    if let result = results.first,
                       result.status == "complete" || result.status == "error" {
                        return result
                    }
                    // status is still "pending", keep waiting
                }
                return nil
            }
        }
    }
    ```

    **Important implementation notes:**

    - Use the global `convex` singleton (from ConvexEnv.swift) — do NOT create a new client
    - The Convex `subscribe(to:with:)` returns a publisher. Use `.values` for async iteration.
    - The subscription to `getResult` returns a single document (or empty). The `for await` loop receives updates as the document changes from "pending" → "complete"/"error"
    - **Timeout helper:** Create a private `withTimeout` helper that races the subscription against a 10-second `Task.sleep`. If timeout wins, cancel the subscription task and return nil.

    ```swift
    private func withTimeout<T>(seconds: Double, operation: @escaping () async -> T?) async -> T? {
        await withTaskGroup(of: T?.self) { group in
            group.addTask { await operation() }
            group.addTask {
                try? await Task.sleep(nanoseconds: UInt64(seconds * 1_000_000_000))
                return nil
            }
            // First result wins
            if let result = await group.next() ?? nil {
                group.cancelAll()
                return result
            }
            group.cancelAll()
            return nil
        }
    }
    ```

    - The `subscribe(to:with:)` call uses `with:` parameter format matching ConvexMobile 0.8.0. The args dict must be `[String: ConvexEncodable?]`.

    Avoid:
    - Do NOT make NLParseService an ObservableObject — views manage their own @State
    - Do NOT create a long-running subscription — subscribe on demand, cancel after result
    - Do NOT import or depend on ChatViewModel — parse is independent of chat

    ```swift
        guard !nlInput.trimmingCharacters(in: .whitespaces).isEmpty else { return }
        isParsing = true

        Task {
            let result = await NLParseService.shared.parse(text: nlInput, type: "event")

            if let fields = result?.eventFields, result?.status == "complete" {
                // Loom parse succeeded — populate form
                title = fields.title
                isAllDay = fields.isAllDay ?? false

                if let startDate = ISO8601DateFormatter().date(from: fields.start) ??
                    parseFlexibleDate(fields.start) {
                    eventDate = startDate
                    startTime = startDate
                    let mins = fields.duration ?? 60
                    endTime = startDate.addingTimeInterval(Double(mins) * 60)
                }
            } else {
                // Fallback to local NLEventParser
                let parsed = NLEventParser.parse(nlInput)
                title = parsed.title
                if let date = parsed.date {
                    eventDate = date
                    startTime = date
                    endTime = date.addingTimeInterval(3600)
                    isAllDay = !parsed.hasTime
                }
            }

            isParsing = false
        }
    }
    ```

    ```swift
    private func parseFlexibleDate(_ string: String) -> Date? {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "en_US_POSIX")
        // Try common formats the LLM might return
        for format in ["yyyy-MM-dd'T'HH:mm:ss", "yyyy-MM-dd'T'HH:mm", "yyyy-MM-dd"] {
            formatter.dateFormat = format
            if let date = formatter.date(from: string) { return date }
        }
        return nil
    }
    ```
