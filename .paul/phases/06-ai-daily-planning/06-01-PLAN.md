<objective>
## Goal
Add daily planning support to the bridge and Swift model/ViewModel layers so Loom can propose a structured daily schedule and the app can approve, reject, or undo the entire plan as a batch.

## Purpose
Phase 6 success criteria require "plan my day" → structured recommendation → batch approve/reject. This plan builds the backend and logic layers. Plan 06-02 will add the UI card and ChatView integration.

## Output
- Bridge understands daily planning requests and outputs `daily_plan` ACTION blocks
- Swift model decodes plan proposals with typed `PlannedBlock` array
- ChatViewModel can batch-create events from a plan, reject plans, and batch-undo
</objective>

    1. Create new file `LoomCal/Models/DailyPlanProposal.swift`:

       ```swift
       import Foundation

       /// A daily plan proposal from Loom containing multiple proposed time blocks.
       /// Decoded from the ChatMessage.action JSON string when type is "daily_plan".
       struct DailyPlanProposal: Codable {
           let type: String              // "daily_plan"
           let displaySummary: String
           let payload: PlanPayload

           struct PlanPayload: Codable {
               let blocks: [PlannedBlock]
           }
       }

       /// A single proposed time block within a daily plan.
       struct PlannedBlock: Codable, Identifiable {
           var id: String { "\(title)-\(start)" }  // Computed from content for stability

           let title: String
           let start: Int       // UTC milliseconds (bridge normalizes from ISO)
           let duration: Int    // minutes

           // Computed helpers
           var startDate: Date {
               Date(timeIntervalSince1970: TimeInterval(start) / 1000)
           }

           var endDate: Date {
               startDate.addingTimeInterval(TimeInterval(duration) * 60)
           }

           var durationFormatted: String {
               if duration < 60 { return "\(duration) min" }
               if duration % 60 == 0 { return "\(duration / 60) hr" }
               return "\(duration / 60) hr \(duration % 60) min"
           }
       }
       ```

       Note: PlannedBlock uses `Int` (not `@ConvexInt`) because these values come from
       the bridge-normalized ACTION JSON, not directly from Convex subscriptions.
       The bridge converts ISO strings → UTC ms strings, and Swift's Codable
       decodes the string numbers as Int.

       However, test both Int and String decoding for `start` and `duration` since
       the bridge outputs String values (String("1708603200000")) but the model expects Int.
       If needed, add custom Codable init that handles both String-encoded and raw Int:
       ```swift
       init(from decoder: Decoder) throws {
           let container = try decoder.container(keyedBy: CodingKeys.self)
           title = try container.decode(String.self, forKey: .title)
           // Handle both String and Int for start/duration (bridge normalizes to strings)
           if let intVal = try? container.decode(Int.self, forKey: .start) {
               start = intVal
           } else if let strVal = try? container.decode(String.self, forKey: .start),
                     let parsed = Int(strVal) {
               start = parsed
           } else {
               throw DecodingError.dataCorruptedError(forKey: .start, in: container,
                   debugDescription: "start must be Int or numeric String")
           }
           if let intVal = try? container.decode(Int.self, forKey: .duration) {
               duration = intVal
           } else if let strVal = try? container.decode(String.self, forKey: .duration),
                     let parsed = Int(strVal) {
               duration = parsed
           } else {
               throw DecodingError.dataCorruptedError(forKey: .duration, in: container,
                   debugDescription: "duration must be Int or numeric String")
           }
       }
       ```

    2. Add `approveDailyPlan(_ message: ChatMessage)`:
       ```swift
       func approveDailyPlan(_ message: ChatMessage) {
           guard message.isDailyPlan,
                 let plan = message.decodedPlan,
                 let calVM = calendarViewModel
           else { return }

           Task {
               do {
                   var createdIds: [String] = []
                   let beforeIds = Set(calVM.events.map { $0._id })

                   // Create each proposed time block as an event
                   for block in plan.payload.blocks {
                       let startDate = block.startDate
                       try await calVM.createEvent(
                           title: block.title,
                           start: startDate,
                           durationMinutes: block.duration,
                           isAllDay: false
                       )
                   }

                   // Wait for subscription to reflect new events, then collect IDs
                   try? await Task.sleep(for: .seconds(2))
                   let afterIds = Set(calVM.events.map { $0._id })
                   createdIds = Array(afterIds.subtracting(beforeIds))

                   // Mark confirmed
                   let statusArgs: [String: ConvexEncodable?] = [
                       "id": message._id,
                       "actionStatus": "confirmed"
                   ]
                   try await convex.mutation("chatMessages:updateActionStatus", with: statusArgs)

                   // Navigate to today and flash highlights
                   calVM.navigateToDate(Date())
                   for id in createdIds {
                       calVM.flashHighlight(eventId: id)
                   }

                   // Start batch undo timer
                   let context = UndoContext(messageId: message._id, createdIds: createdIds)
                   self.startUndoTimer(for: context)

               } catch {
                   let failArgs: [String: ConvexEncodable?] = [
                       "id": message._id,
                       "actionStatus": "cancelled"
                   ]
                   try? await convex.mutation("chatMessages:updateActionStatus", with: failArgs)

                   let errorArgs: [String: ConvexEncodable?] = [
                       "role": "assistant",
                       "content": "I couldn't create the schedule — please try again."
                   ]
                   try? await convex.mutation("chatMessages:send", with: errorArgs)
               }
           }
       }
       ```
