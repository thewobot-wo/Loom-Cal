```swift
struct LoomEvent: Decodable {
    let _id: String
    let calendarId: String
    let title: String
    @ConvexInt var start: Int          // UTC milliseconds — v.int64()
    @ConvexInt var duration: Int       // minutes — v.int64()
    let timezone: String
    let isAllDay: Bool
    let location: String?
    let notes: String?                 // markdown plain text
    let url: String?                   // dedicated meeting link field
    let color: String?
    let rrule: String?                 // RRULE recurrence string
    let recurrenceGroupId: String?
    let attachments: [String]?
}
```

**LoomCal/Models/StudioEvent.swift:**
```swift
import ConvexMobile

struct StudioEvent: Decodable {
    let _id: String
    let calendarId: String
    let title: String
    @ConvexInt var start: Int          // UTC milliseconds — v.int64()
    @ConvexInt var duration: Int       // minutes — v.int64()
    let timezone: String
    let isAllDay: Bool
    @ConvexInt var lastSyncedAt: Int   // UTC milliseconds — v.int64()
}
```

CRITICAL for all structs:
- Every `v.int64()` field MUST use `@ConvexInt var` (not `let`)
- Every `v.optional(v.int64())` field MUST use `@OptionalConvexInt var` (if available) or handle manually
- All `@ConvexInt` fields MUST be `var`, not `let` — property wrappers require `var`
- Optional fields in schema (`v.optional(...)`) map to Swift optionals (`Type?`)

If `@OptionalConvexInt` is not available in the ConvexMobile version, use a manual CodingKeys approach to handle optional int64 fields. Check `https://docs.convex.dev/client/swift/data-types` for the correct pattern.

**LoomCal/Views/ContentView.swift:**
Create a simple proof-of-concept view that subscribes to `events:list` and displays results:

```swift
import SwiftUI
import ConvexMobile

struct ContentView: View {
    @State private var events: [LoomEvent] = []
    @State private var connectionStatus: String = "Connecting..."

    var body: some View {
        NavigationStack {
            VStack {
                Text(connectionStatus)
                    .font(.caption)
                    .foregroundStyle(.secondary)

                List(events, id: \._id) { event in
                    VStack(alignment: .leading) {
                        Text(event.title)
                            .font(.headline)
                        Text("Start: \(event.start)")
                            .font(.caption)
                    }
                }
            }
            .navigationTitle("Loom Cal")
        }
        .task {
            connectionStatus = "Connected"
            for await result: [LoomEvent] in convex
                .subscribe(to: "events:list")
                .replaceError(with: [])
                .values
            {
                self.events = result
            }
        }
    }
}
```

This view proves: (1) ConvexClient connects, (2) subscription works, (3) LoomEvent decodes correctly including @ConvexInt fields.
