**LoomCal/Models/LoomAction.swift** (new file) — Create a Codable struct:

```swift
import Foundation

struct LoomAction: Codable, Equatable {
    let type: String          // "create_event", "update_event", "delete_event", "create_task", "update_task", "delete_task"
    let displaySummary: String // Human-readable summary for confirmation card
    let payload: [String: ActionValue]
    let previousValues: [String: ActionValue]?  // For edits — shows what changed

    // Flexible value type since payload fields can be String, Int, Bool
    enum ActionValue: Codable, Equatable {
        case string(String)
        case int(Int)
        case bool(Bool)

        init(from decoder: Decoder) throws {
            let container = try decoder.singleValueContainer()
            if let boolVal = try? container.decode(Bool.self) {
                self = .bool(boolVal)
            } else if let intVal = try? container.decode(Int.self) {
                self = .int(intVal)
            } else if let strVal = try? container.decode(String.self) {
                self = .string(strVal)
            } else {
                throw DecodingError.dataCorruptedError(in: container, debugDescription: "Cannot decode ActionValue")
            }
        }

        func encode(to encoder: Encoder) throws {
            var container = encoder.singleValueContainer()
            switch self {
            case .string(let v): try container.encode(v)
            case .int(let v): try container.encode(v)
            case .bool(let v): try container.encode(v)
            }
        }

        var stringValue: String? {
            if case .string(let v) = self { return v }
            return nil
        }

        var intValue: Int? {
            if case .int(let v) = self { return v }
            if case .string(let v) = self { return Int(v) }
            return nil
        }

        var boolValue: Bool? {
            if case .bool(let v) = self { return v }
            return nil
        }
    }

    // Convenience: check if action is event-related vs task-related
    var isEventAction: Bool {
        type.contains("event")
    }

    var isTaskAction: Bool {
        type.contains("task")
    }

    var isCreate: Bool { type.hasPrefix("create_") }
    var isUpdate: Bool { type.hasPrefix("update_") }
    var isDelete: Bool { type.hasPrefix("delete_") }
}
```

BUT we need to be careful: ChatViewModel is `@MainActor`, and so are CalendarViewModel and TaskViewModel, so direct property access is fine.

After mutation, call `calendarViewModel?.flashHighlight(eventId:)` and navigate to the event's date.
   - `"update_event"`: Extract id from payload, build update args from payload fields. Call `calendarViewModel?.updateEvent(...)`. Flash highlight.

Add `startUndoTimer(for action: LoomAction, messageId: String, createdId: String?)`:
1. Set `activeUndoAction = (action, messageId, createdId)`
2. Set `undoSecondsRemaining = 5`
3. Cancel existing undoTask

Add `undoAction()`:

2. Cancel undoTask, clear activeUndoAction
3. Mark action as undone: `convex.mutation("chatMessages:updateActionStatus", with: ["id": undo.messageId, "actionStatus": "undone"])`
4. Reverse: if create_event/create_task and createdId exists, delete it. If update_event/update_task and previousValues exist, patch back. Skip undo for deletes (data is gone — too complex for Phase 5, just don't start undo timer for deletes).

IMPORTANT: For the `id` parameter in update/delete calls, the Convex document ID is a string. The existing ViewModel methods accept it as the ConvexMobile ID type.
